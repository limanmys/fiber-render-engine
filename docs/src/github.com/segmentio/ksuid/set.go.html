func varintLength64( uint64) int {

	case ( & 0xFFFFFFFFFFFFFF00) == 0:
		return 1

		return 2
	case ( & 0xFFFFFFFFFF000000) == 0:
		return 3
	case ( & 0xFFFFFFFF00000000) == 0:
		return 4
	case ( & 0xFFFFFF0000000000) == 0:
		return 5
	case ( & 0xFFFF000000000000) == 0:
		return 6

		return 7

	}
