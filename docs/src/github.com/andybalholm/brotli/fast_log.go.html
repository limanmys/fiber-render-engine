	return uint32(bits.Len()) - 1
}

/* Faster logarithm for small integers, with the property of log2(0) == 0. */
func fastLog2( uint) float64 {
	if  < uint(len(kLog2Table)) {
		return float64(kLog2Table[])
	}

	return math.Log2(float64())
}
