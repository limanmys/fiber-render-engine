func toUpperCase( []byte) int {
	if [0] < 0xC0 {
		if [0] >= 'a' && [0] <= 'z' {
			[0] ^= 32
		}

		return 1
	}

	/* An overly simplified uppercasing model for UTF-8. */
	if [0] < 0xE0 {
		[1] ^= 32
		return 2
	}

	/* An arbitrary transform for three byte characters. */
	[2] ^= 5

	return 3
}
