	if [].freq == [].freq && [].literal < [].literal || [].freq < [].freq {
		[], [] = [], []

		[], [] = [], []

		if [].freq == [].freq && [].literal < [].literal || [].freq < [].freq {
			[], [] = [], []
		}
	}

}
