		return .unlink(.Filepath)

	case "Link":
		return .link(.Filepath, .Target)

	case "Symlink":
		// NOTE: r.Filepath is the target, and r.Target is the linkpath.
		return .symlink(.Filepath, .Target)
	}

	return errors.New("unsupported")
