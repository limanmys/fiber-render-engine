func ( *bufPool) () []byte {
	if .blen <= 0 {
		panic("bufPool: new buffer creation length must be greater than zero")
	}

	for {
		select {
		case  := <-.ch:
			if cap() < .blen {
				// just in case: throw away any buffer with insufficient capacity.
				continue
			}

			return [:.blen]

		default:
			return make([]byte, .blen)
		}
	}
}
