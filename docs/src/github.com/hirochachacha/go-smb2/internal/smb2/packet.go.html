func ( PacketCodec) () bool {
	if len() < 64 {
		return true
	}

	 := .ProtocolId()
	if [0] != 0xfe {
		return true
	}
	if [1] != 'S' {
		return true
	}
	if [2] != 'M' {
		return true
	}
	if [3] != 'B' {
		return true
	}

	if .StructureSize() != 64 {
		return true
	}

	if .NextCommand()&7 != 0 {
		return true
	}

	return false
}
