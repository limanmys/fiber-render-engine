func respUnauthorizedNegotiate( *http.Response) bool {
	if .StatusCode == http.StatusUnauthorized {
		if .Header.Get(HTTPHeaderAuthResponse) == HTTPHeaderAuthResponseValueKey {
			return true
		}
	}
	return false
}
