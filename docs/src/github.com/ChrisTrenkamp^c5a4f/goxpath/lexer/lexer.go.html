func findOperatorState( *Lexer) stateFn {
	.skipWS(true)

	switch string(.peek()) {
	case ">", "<", "!":
		.next()
		if string(.peek()) == "=" {
			.next()
		}
		.emit(XItemOperator)
		return startState
	case "|", "+", "-", "*", "=":
		.next()
		.emit(XItemOperator)
		return startState
	case "(":
		.next()
		.emit(XItemOperator)
		for  := startState;  != nil; {
			 = ()
		}
		.skipWS(true)
		if string(.next()) != ")" {
			return .errorf("Missing end )")
		}

		.next()
