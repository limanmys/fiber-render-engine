// Config configuration for now package
type Config struct {
	WeekStartDay time.Weekday
	TimeLocation *time.Location
	TimeFormats  []string
}

// DefaultConfig default config
var DefaultConfig *Config

// New initialize Now based on configuration
func ( *Config) ( time.Time) *Now {
	return &Now{Time: , Config: }
}

// Parse parse string to time based on configuration
func ( *Config) ( ...string) (time.Time, error) {
	if .TimeLocation == nil {
		return .With(time.Now()).Parse(...)
	} else {
		return .With(time.Now().In(.TimeLocation)).Parse(...)
	}
}

// MustParse must parse string to time or will panic
func ( *Config) ( ...string) time.Time {
	if .TimeLocation == nil {
		return .With(time.Now()).MustParse(...)
	} else {
		return .With(time.Now().In(.TimeLocation)).MustParse(...)
	}
}
