// Variant returns the variant encoded in uuid.
func ( UUID) () Variant {
	switch {
	case ([8] & 0xc0) == 0x80:
		return RFC4122
	case ([8] & 0xe0) == 0xc0:
		return Microsoft
	case ([8] & 0xe0) == 0xe0:
		return Future
	default:
		return Reserved
	}
}
