package msgp

// The sizes provided
// are the worst-case
// encoded sizes for
// each type. For variable-
// length types ([]byte, string),
// the total encoded size is
// the prefix size plus the
// length of the object.

	UintSize       = Int64Size

	Int16Size      = 3
	Int32Size      = 5

	Uint16Size     = 3
	Uint32Size     = 5

	Float32Size    = 5
