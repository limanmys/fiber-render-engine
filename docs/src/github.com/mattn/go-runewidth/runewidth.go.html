	}

		switch {
		case  < 0x20:
			return 0
		case ( >= 0x7F &&  <= 0x9F) ||  == 0xAD: // nonprint
			return 0
		case  < 0x300:
			return 1
		case inTable(, narrow):
			return 1
		case inTables(, nonprint, combining):
			return 0
		case inTable(, doublewidth):
			return 2
		default:
			return 1
		}
	} else {
		switch {
		case inTables(, nonprint, combining):
			return 0
		case inTable(, narrow):
			return 1
		case inTables(, ambiguous, doublewidth):
			return 2
		case !.StrictEmojiNeutral && inTables(, ambiguous, emoji, narrow):
			return 2
		default:
			return 1
		}
	}
}
