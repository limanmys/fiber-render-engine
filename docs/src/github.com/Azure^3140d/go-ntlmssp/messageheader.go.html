package ntlmssp

var signature = [8]byte{'N', 'T', 'L', 'M', 'S', 'S', 'P', 0}

type messageHeader struct {
	Signature   [8]byte
	MessageType uint32
}

func ( messageHeader) () bool {
	return bytes.Equal(.Signature[:], signature[:]) &&
		.MessageType > 0 && .MessageType < 4
}

func newMessageHeader( uint32) messageHeader {
	return messageHeader{signature, }
}
