func ( *gorm.DB) {
	if !.Config.SkipDefaultTransaction && .Error == nil {
		if  := .Begin(); .Error == nil {
			.Statement.ConnPool = .Statement.ConnPool
			.InstanceSet("gorm:started_transaction", true)
		} else if .Error == gorm.ErrInvalidTransaction {
			.Error = nil
		} else {
			.Error = .Error
		}
	}
}

			if .Error != nil {
				.Rollback()
			} else {
				.Commit()
			}

			.Statement.ConnPool = .ConnPool
		}
	}
